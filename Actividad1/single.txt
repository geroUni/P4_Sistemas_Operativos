def is\_palindrome(word):
"""
This function checks if a given word is a palindrome.
A palindrome is a word that reads the same backward as forward.

:param str word: The word to check for palindromeness
:return: True if the word is a palindrome, False otherwise
:rtype: bool
"""

# Remove any non-alphanumeric characters and convert to lowercase
processed\_word = ''.join(c.lower() for c in word if c.isalnum()).strip()

# Check if the processed word is the same as its reverse
return processed\_word == processed\_word[::-1]
